package com.tgwhietstudio.tgwai; import android.app.Activity; import android.os.Bundle; import android.content.Intent; import android.speech.RecognizerIntent; import android.widget.Toast; import java.util.ArrayList; public class MainActivity extends Activity { @Override protected void onCreate(Bundle b) { super.onCreate(b); Intent i = new Intent(RecognizerIntent.ACTION_RECOGNIZE_SPEECH); i.putExtra(RecognizerIntent.EXTRA_LANGUAGE, "vi-VN"); startActivityForResult(i, 1); } @Override protected void onActivityResult(int r, int c, Intent d) { super.onActivityResult(r, c, d); if (r == 1 && c == RESULT_OK && d != null) { ArrayList t = d.getStringArrayListExtra(RecognizerIntent.EXTRA_RESULTS); if (t != null && t.size() > 0) { String s = t.get(0).toLowerCase(); Toast.makeText(this, s, Toast.LENGTH_LONG).show(); if (s.contains("youtube")) { startActivity(new Intent(Intent.ACTION_VIEW, android.net.Uri.parse("https://youtube.com"))); } } } } }
